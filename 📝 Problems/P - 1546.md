#include <algorithm>

#include <iostream>

auto main() -> int {
  std::cin.tie(nullptr)->sync_with_stdio(false);

  int n;
  double sum = 0, avg;
  if (std::cin >> n) {
    std::vector<double> v(n);
    for (double &x : v) {
      std::cin >> x;
    }
    std::ranges::sort(v);
    double m = std::ranges::max(v);
    for (auto _ : v) {
      sum = std::accumulate(v.begin(), v.end(), 0.0);
    }
    avg = (sum / m * 100) / n;
    std::cout << std::fixed << std::setprecision(6) << avg;
  }
  return 0;
}
