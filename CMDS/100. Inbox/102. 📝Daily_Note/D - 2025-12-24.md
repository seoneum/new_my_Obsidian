const todayStr = "2025-12-24";
const ydayStr  = "2025-12-23";

// created 값이 어떤 타입이든 안전하게 yyyy-MM-dd 문자열로 변환
const toYMD = (v) => {
  if (!v) return null;

  // 이미 yyyy-mm-dd 문자열이면 그대로(또는 앞 10글자)
  if (typeof v === "string") return v.slice(0, 10);

  // Luxon DateTime (Dataview가 자주 쓰는 타입)
  if (v && typeof v.toFormat === "function") return v.toFormat("yyyy-MM-dd");

  // JS Date
  if (v instanceof Date) return dv.luxon.DateTime.fromJSDate(v).toFormat("yyyy-MM-dd");

  // Dataview가 가끔 {year,month,day} 형태 object로 주는 경우
  if (typeof v === "object" && v.year && v.month && v.day) {
    return dv.luxon.DateTime.fromObject({ year: v.year, month: v.month, day: v.day }).toFormat("yyyy-MM-dd");
  }

  // 그 외는 실패 처리
  return null;
};

const pages = dv.pages().where(p => p.file.path !== dv.current().file.path);

const pagesToday = pages.where(p => toYMD(p.created) === todayStr);
const pagesYday  = pages.where(p => toYMD(p.created) === ydayStr);

// tagging/needed 카운트
const needTagging = dv.pages().where(p => Array.isArray(p.tags) && p.tags.includes("tagging/needed"));

dv.paragraph(`- Notes created today: **${pagesToday.length}**`);
dv.paragraph(`- Notes created yesterday: **${pagesYday.length}**`);
dv.paragraph(`- tagging/needed remaining: **${needTagging.length}**`);
